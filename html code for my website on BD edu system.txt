        function loadOpinions() {
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) {
                try {
                    const parsed = JSON.parse(stored);
                    if (Array.isArray(parsed) && parsed.length > 0) {

                    }
                } catch(e) {}
            } else {
                saveOpinions();
            }

        function saveOpinions() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(opinions));
        }

        function formatTime(timestamp) {
            const diff = Date.now() - timestamp;
            const minutes = Math.floor(diff / 60000);
            const hours = Math.floor(diff / 3600000);
            const days = Math.floor(diff / 86400000);
            if (minutes < 1) return "just now";
            if (minutes < 60) return `${minutes} min ago`;
            if (hours < 24) return `${hours} hour${hours > 1 ? 's' : ''} ago`;
            return `${days} day${days > 1 ? 's' : ''} ago`;
        }

        function escapeHtml(str) {
            if (!str) return "";
            return str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;').replace(/'/g, '&#39;');
        }

        function renderOpinions() {
            const container = document.getElementById("opinionList");
            if (!container) return;
            if (!opinions || opinions.length === 0) {
                container.innerHTML = `<div style="text-align:center;padding:2rem;color:#8b7f6e;">No opinions yet. Be the first to speak up.</div>`;
                return;
            }
            const sorted = [...opinions].sort((a, b) => b.timestamp - a.timestamp);

                        <div class="opinion-time">📅 ${formatTime(op.timestamp)}</div>
                    </div>
                `;
            }

        }
